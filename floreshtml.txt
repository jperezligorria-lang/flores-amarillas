/* 🌻 FLORES */
const arbol = document.querySelector(".arbol");

for(let i=0;i<120;i++){
  let flor=document.createElement("div");
  flor.className="flor";

  let x=Math.random()*200;
  let y=Math.random()*160;

  flor.style.left=x+"px";
  flor.style.top=y+"px";

  arbol.appendChild(flor);
}

/* ⏱ CONTADOR DESDE 8 MARZO 2023 */
const inicio = new Date("2023-03-08T00:00:00");

function actualizar(){
  const ahora = new Date();
  const diff = ahora - inicio;

  const dias = Math.floor(diff/(1000*60*60*24));
  const horas = Math.floor(diff/(1000*60*60)%24);
  const minutos = Math.floor(diff/(1000*60)%60);
  const segundos = Math.floor(diff/1000%60);

  document.getElementById("tiempo").innerHTML =
    dias+" días "+
    horas+" horas "+
    minutos+" minutos "+
    segundos+" segundos";
}

setInterval(actualizar,1000);
actualizar();
